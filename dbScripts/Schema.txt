CREATE TABLE sys_chips (
	id serial NOT NULL,
	x int4 NULL,
	y int4 NULL,
	rot int4 NULL,
	sys_id int4 NULL,
	cid int4 NULL,
	"createdAt" timestamptz NOT NULL,
	"updatedAt" timestamptz NOT NULL,
	s_x int4 NULL,
	s_y int4 NULL,
	"type" varchar(255) NULL,
	is_shown_to_developer bool NULL,
	t_x int4 NULL,
	t_y int4 NULL,
	rdl_snaped bool NOT NULL DEFAULT false,
	CONSTRAINT sys_chips_pkey PRIMARY KEY (id),
	CONSTRAINT sys_chips_chip_id_fkey FOREIGN KEY (cid) REFERENCES chips(id),
	CONSTRAINT sys_chips_sys_id_fkey FOREIGN KEY (sys_id) REFERENCES sys(id)
);
